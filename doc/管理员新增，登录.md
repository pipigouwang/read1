1.管理员新增：
http://www.huahong.com/admin/users/rulegroup

~~~
{
	"tag":"user",
	"op":"user_create",
	"data":{
		"account":"admin",//账号
		"password":"admin",
		"name":"huahua"//用户姓名
	}
}
返回：
{
    "message": "success",
    "err": 2014,
    "data": {
        "user_id": "4"//管理员id  添加权限时用
    }
}
~~~
2.管理员修改（包括修改状态)
~~~
{
	"tag":"user",
	"op":"user_edit",
	"data":{
		"id":"5",
		"account":"admin",
		"password":"admin",
		"name":"huahua134",
		"state":1 //0删除1，禁用，2启用
	}
}

返回：
  {
       "message": "success",
       "err": 2014,
       "data": null
   }
~~~
