咨讯：
url:域名/admin/news/index
1.1 列表
~~~
{
	"tag":"news",
	"op":"getlist",
	"condition":{
		"status":1	
	}
}

返回
{
    "message": "success",
    "err": 7000,
    "data": {
        "list": [
            {
                "id": 1,
                "title": "afsd"，//标题
                "text": "asdfasdf", //富文本内容
                "create_time": "2018-08-15 10:44:44", //创建时间
                "clicks": 1, //点击量  暂时不做显示
                "status": 1 //状态 0,删除，1，启用，2，禁用
            },
            {
                "id": 2,
                "title": "2",
                "text": "asdfasdfasdf",
                "create_time": "2018-08-08 10:52:44",
                "clicks": null,
                "status": 1
            }
        ],
        "total": 2
    }
}
~~~
1.2 新增
~~~
{
	"tag":"news",
	"op":"add",
	"data":{
		"title":2,
		"status":1,
		"text":"asdfasdfasdf"
	}
}

~~~
1.3 修改

~~~
{
	"tag":"news",
	"op":"update",
	"condition":{
		"id":1	
	},
	"data":{
		"title":2,
		"status":1,
		"text":"asdfasdfasdf"
	}
}
~~~
1.4 咨讯详情 
~~~
{
	"tag":"news",
	"op":"getone",
	"condition":{
		"id":2	
	}
}
~~~

1.5 咨讯状态修改

~~~
{
	"tag":"news",
	"op":"state",
	"condition":{
		"id":2	
	},
	"data":{
		"status":2
	}
}
~~~
